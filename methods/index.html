# new jsMap(order, yaxis,angle, constructor, callback )

  • 描述： 初始化地图 注：获取到设备数据以及地图dom容器存在于dom树中后方可调用

  • 参数：

    • order：
      • 描述： 当前设备所处楼层order
      • 类型： Number
    • yaxis：
      • 描述： 当前设备点位
      • 类型： Number
    • angle：
      • 描述： 当前设备角度
      • 类型： Number
    • constructor：
      • 描述： 配置文件的实例
      • 类型： constructor构造函数
    • callback：
      • 描述： 地图初始化加载完成之后的回调
      • 类型： Function
    //map 即在middleware.js中定义的变量
map = new jsMap(
    2,
    34,
    45,
    new Config(),
    function () {
        eval(map.options.initFinish)
    }
)

# changeFloor(floorOrder, Boolean, callback)

  • 描述： 切换楼层调用函数
  • 参数：
    • floorOrder:
      • 描述： 切换到第几层的order值
      • 类型： Number
    • Boolean:
      • 描述： 是否在寻路中
      • 类型： Boolean
    • callback:
      • 描述： 切换成功之后的回调
      • 类型： Function
map.changeFloor(3, false, () => {})

# facilities(type)

  • 描述： 公共设施 返回离当前设备最近的楼层order
  • 参数：
    • type:
      • 描述： 公共设施type值 具体联系开发人员
      • 类型： Number
    const order = map.facilities(5)

# findFacilities(order)

  • 描述： 地图调用方法 需在对应业务代码中挂载至全局变量或者window全局中
  • 参数：
    • order:
      • 描述： 参数为地图回传 距离设施最近的楼层order
 //公共设施导航 index为地图回传的楼层order
 //此方法即为QmMap-config.js里配置的方法 需在created阶段 或点击之前挂载至window或其他地图内部能访问到的全局变量上
!window.findFacilities && (window.findFacilities = this.findFacilities)

findFacilities(index) {
    //找到公共设施路径
    let findFacilities = map[map.options.defaultSet.findFacilities.findObj][index]
    //判断返回的路径里是否有值
    if (findFacilities.path.niceMode) {
        let returnPath = findFacilities.path.niceMode
        //调用map.go 开始导航至公共设施
        map.go(returnPath, function () {
            console.log('前端寻路结束!')
        })
    }
}

# findPath(order, yaxis, shopOrder，shopYaxis)

  • 描述： 店铺导航所有路径

  • 参数：

    • order:
      • 描述： 当前设备所属楼层order
      • 类型： Number
    • yaxis:
      • 描述： 当前设备所属楼层点位
      • 类型： Number
    • shopOrder:
      • 描述： 目标店铺的楼层order
      • 类型： Number
    • shopYaxis:
      • 描述： 目标店铺的导航点
      • 类型： Number

  • 返回值：

    {
	niceMode：{...} //最佳导航路径
	stairMode: {...} //扶梯导航路径
	liftMode: {...} //电梯导航路径
}

# go(path, callback)

  • 描述： 导航
  • 参数：
    • path：
      • 描述： 导航路径
      • 类型： Object
    • callback：
      • 描述： 导航完成之后的回调
      • 类型： Function
map.go(this.path, function () {})

# speedUp()

  • 描述： 人物模型或导航图标加速 初次调用为加速 二次调用为恢复初始速度

# pause(Boolean)

  • 描述： 控制人物模型或者导航图标是否暂停
  • 参数：
    • Boolean：
      • **描述：**为true时停止导航 false时开始导航
      • 类型： Boolean
